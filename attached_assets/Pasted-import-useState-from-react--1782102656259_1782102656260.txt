// ── Statistical fallback predictor (no API needed) ────────────────
function fallbackPred(home, away) {
  const h = td(home), a = td(away);
  const formPts = s => [...s].reduce((acc, c, i) => acc + (c==="W"?3:c==="D"?1:0) * (0.85**(4-i)), 0);
  const rankEdge = Math.min(35, Math.max(-35, (a[1] - h[1]) * 0.45));
  const formEdge = Math.min(12, Math.max(-12, (formPts(h[2]) - formPts(a[2])) * 4));
  const wcEdge   = Math.min(8,  Math.max(-8,  (h[3]-h[4]) - (a[3]-a[4])));
  const raw = 47 + rankEdge + formEdge + wcEdge;
  const homeWin = Math.round(Math.min(75, Math.max(10, raw)));
  const draw    = Math.round(Math.max(12, 28 - Math.abs(raw-47)*0.3));
  const awayWin = Math.max(5, 100 - homeWin - draw);
  const hG = Math.max(0, Math.round((h[3]/(Math.max(h[4],0.5)))*0.8));
  const aG = Math.max(0, Math.round((a[3]/(Math.max(a[4],0.5)))*0.65));
  return {
    homeWinPct: homeWin, drawPct: draw, awayWinPct: awayWin,
    predictedScore: `${hG}–${aG}`,
    keyFactors: [
      `${home} (FIFA #${h[1]}) vs ${away} (FIFA #${a[1]}) — ${h[1]<a[1]?home:away} ranked higher`,
      `Recent form: ${home} ${h[2]} · ${away} ${a[2]}`,
      `WC 2026 scoring: ${home} ${h[3]}GF/${h[4]}GA — ${away} ${a[3]}GF/${a[4]}GA`,
    ],
    confidence: Math.abs(h[1]-a[1]) > 40 ? "High" : "Medium",
    analysis: `Statistical model favors ${homeWin>awayWin?home:awayWin>homeWin?away:"a draw"} based on FIFA rankings, form, and WC 2026 performance. Predicted score: ${hG}–${aG}.`,
    isFallback: true,
  };
}

      )}

      </div>
    </div>
  );
}

function HistCard({ pred }) {

  const fav = homeWinPct>awayWinPct ? match.home : awayWinPct>homeWinPct ? match.away : "Draw";

  return (
    <div style={{ background:K.card, border:`1px solid ${K.border}`, borderRadius:10, padding:"12px 16px", marginBottom:8, display:"flex", alignItems:"center", gap:12 }}>
      <div style={{ fontSize:24 }}>🔮</div>
      <div style={{ flex:1, minWidth:0 }}>
        <div style={{ fontSize:13, fontWeight:600, color:K.text, overflow:"hidden", textOverflow:"ellipsis", whiteSpace:"nowrap" }}>
          {h[0]} {match.home} vs {a[0]} {match.away}
        </div>
        <div style={{ fontSize:11, color:K.dim, marginTop:2 }}>Group {match.group} · {match.date} · {match.time}</div>
        <div style={{ display:"flex", gap:8, marginTop:4 }}>
          <span style={{ fontSize:10, color:K.win }}>{homeWinPct}%</span>
          <span style={{ fontSize:10, color:K.draw }}>{drawPct}%</span>
          <span style={{ fontSize:10, color:K.loss }}>{awayWinPct}%</span>
        </div>
      </div>
      <div style={{ textAlign:"right", flexShrink:0 }}>
        <div style={{ fontSize:16, fontWeight:800, color:K.gold }}>{predictedScore}</div>
        <div style={{ fontSize:10, color:K.dim, marginTop:1 }}>Favors {fav.split(" ")[0]}</div>

        <div style={{ fontSize:10, color:K.dim, marginTop:1 }}>{timestamp}</div>
      </div>
    </div>
  );
}
